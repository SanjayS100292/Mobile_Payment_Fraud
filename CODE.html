# Mobile-Payments Fraud: Isolation Forest vs DBSCAN (end-to-end)
# 1) Set DATA_PATH below to your Kaggle CSV file.
# 2) (Optional) Keep SAMPLE_N small for quick first run.
# 3) Run cells in order; figures + metrics CSV will be saved to ./figures

import os
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.preprocessing import RobustScaler
from sklearn.decomposition import PCA
from sklearn.ensemble import IsolationForest
from sklearn.cluster import DBSCAN
from sklearn.metrics import (
    confusion_matrix, ConfusionMatrixDisplay,
    precision_recall_curve, average_precision_score,
    roc_curve, roc_auc_score,
    precision_score, recall_score, f1_score
)

# --------- USER CONFIG ---------
DATA_PATH = "/content/paysim.csv"  # <-- set this to your CSV path
SAMPLE_N = 500000                       # reduce for speed; set None to use full data
RANDOM_STATE = 42
CONTAMINATION = 0.002                   # ~0.2%; tune via validation
IF_N_ESTIMATORS = 200
DBSCAN_EPS = 2.0                        # tune after k-distance plot
DBSCAN_MIN_SAMPLES = 5
PCA_COMPONENTS = 10                     # for DBSCAN speed/stability

FIG_DIR = "figures"
os.makedirs(FIG_DIR, exist_ok=True)

# ---------- LOAD DATA ----------
df = pd.read_csv(DATA_PATH)
print("Raw shape:", df.shape)
print("Columns:", list(df.columns)[:20], "..." if len(df.columns) > 20 else "")
print(df.head(3))

# ---------- CLEANING & FEATURE ENGINEERING ----------
# Drop string IDs to avoid leakage (if present)
for c in ["nameOrig", "nameDest"]:
    if c in df.columns:
        df.drop(columns=[c], inplace=True)

# Ensure expected columns exist (map names if your CSV differs)
required = ["type", "amount", "oldbalanceOrg", "newbalanceOrig", "oldbalanceDest", "newbalanceDest"]
missing = [c for c in required if c not in df.columns]
if missing:
    raise ValueError(f"Missing required columns: {missing}. Map your CSV columns accordingly.")

# Create balance error features
df["errorBalanceOrig"] = df["oldbalanceOrg"] - df["newbalanceOrig"] - df["amount"]
df["errorBalanceDest"] = df["oldbalanceDest"] + df["amount"] - df["newbalanceDest"]

# Handle Inf/NaN
df.replace([np.inf, -np.inf], np.nan, inplace=True)
df.dropna(inplace=True)

# One-hot encode transaction type
df = pd.get_dummies(df, columns=["type"], drop_first=True)

# Confirm label column
label_cols = [c for c in df.columns if c.lower() in ["isfraud", "is_fraud"]]
if not label_cols:
    raise ValueError("No 'isFraud' label found. Ensure your CSV contains a fraud label column named 'isFraud'.")
LABEL = label_cols[0]

# Optional subsampling for speed
if SAMPLE_N is not None and df.shape[0] > SAMPLE_N:
    df = df.sample(SAMPLE_N, random_state=RANDOM_STATE).reset_index(drop=True)

print("Post-clean shape:", df.shape)
fraud_rate = df[LABEL].mean()
print(f"Fraud prevalence: {fraud_rate:.6f} ({fraud_rate*100:.4f}%)")

# Robust scaling for skewed features
scale_cols = ["amount","oldbalanceOrg","newbalanceOrig","oldbalanceDest","newbalanceDest",
              "errorBalanceOrig","errorBalanceDest"]
scale_cols = [c for c in scale_cols if c in df.columns]
scaler = RobustScaler()
df[scale_cols] = scaler.fit_transform(df[scale_cols])

# Feature/target split
y = df[LABEL].astype(int)
X = df.drop(columns=[LABEL])
if "isFlaggedFraud" in X.columns:
    X = X.drop(columns=["isFlaggedFraud"])

print("X shape:", X.shape, "| y shape:", y.shape)

# ---------- UTILITIES ----------
def preds_from_top_q(scores, q=0.2):
    """Mark top q% highest scores as anomalies (1), rest 0."""
    k = max(1, int(len(scores) * q / 100.0))
    idx = np.argsort(scores)[::-1]
    y_pred = np.zeros(len(scores), dtype=int)
    y_pred[idx[:k]] = 1
    return y_pred

def precision_at_k(y_true, scores, k):
    k = min(k, len(scores))
    idx = np.argsort(scores)[::-1][:k]
    return y_true.iloc[idx].sum() / k

def precision_at_k_curve(y_true, scores, ks=(50, 100, 250, 500, 1000, 2000, 5000)):
    precs = []
    for k in ks:
        k = min(k, len(scores))
        idx = np.argsort(scores)[::-1][:k]
        precs.append(y_true.iloc[idx].sum() / k)
    plt.figure()
    plt.plot(list(ks), precs, marker="o")
    plt.title("Precision@k")
    plt.xlabel("k (top-k flagged)")
    plt.ylabel("Precision@k")
    plt.grid(True)
    out = os.path.join(FIG_DIR, "precision_at_k_curve.png")
    plt.savefig(out, bbox_inches="tight", dpi=150)
    plt.show()
    print("Saved:", out)

def plot_iso_score_histogram(iso_scores, bins=50):
    plt.figure()
    plt.hist(iso_scores, bins=bins)
    plt.title("Isolation Forest anomaly-score distribution")
    plt.xlabel("Anomaly score (higher = more anomalous)")
    plt.ylabel("Count")
    out = os.path.join(FIG_DIR, "iso_score_hist.png")
    plt.savefig(out, bbox_inches="tight", dpi=150)
    plt.show()
    print("Saved:", out)

def plot_pr_curve(y_true, scores, title_prefix="Model"):
    precision, recall, _ = precision_recall_curve(y_true, scores)
    ap = average_precision_score(y_true, scores)
    plt.figure()
    plt.plot(recall, precision)
    plt.title(f"{title_prefix} PrecisionRecall (AP={ap:.4f})")
    plt.xlabel("Recall")
    plt.ylabel("Precision")
    plt.grid(True)
    out = os.path.join(FIG_DIR, f"{title_prefix.lower().replace(' ', '_')}_pr.png")
    plt.savefig(out, bbox_inches="tight", dpi=150)
    plt.show()
    print("Saved:", out)

def plot_roc(y_true, scores, title_prefix="Model"):
    fpr, tpr, _ = roc_curve(y_true, scores)
    auc = roc_auc_score(y_true, scores)
    plt.figure()
    plt.plot(fpr, tpr)
    plt.title(f"{title_prefix} ROC (AUC={auc:.4f})")
    plt.xlabel("False Positive Rate")
    plt.ylabel("True Positive Rate")
    plt.grid(True)
    out = os.path.join(FIG_DIR, f"{title_prefix.lower().replace(' ', '_')}_roc.png")
    plt.savefig(out, bbox_inches="tight", dpi=150)
    plt.show()
    print("Saved:", out)

def plot_confusion(y_true, y_pred, title="Confusion Matrix", fname="confusion.png"):
    cm = confusion_matrix(y_true, y_pred)
    disp = ConfusionMatrixDisplay(confusion_matrix=cm)
    plt.figure()
    disp.plot(values_format="d")
    plt.title(title)
    out = os.path.join(FIG_DIR, fname)
    plt.savefig(out, bbox_inches="tight", dpi=150)
    plt.show()
    print("Saved:", out)

def metric_bars(y_true, y_pred_a, y_pred_b, name_a="IsolationForest", name_b="DBSCAN", title="Model comparison (Precision/Recall/F1)"):
    def metrics(y_t, y_p):
        return (
            precision_score(y_t, y_p, zero_division=0),
            recall_score(y_t, y_p, zero_division=0),
            f1_score(y_t, y_p, zero_division=0)
        )
    p_a, r_a, f_a = metrics(y_true, y_pred_a)
    p_b, r_b, f_b = metrics(y_true, y_pred_b)

    labels = ["Precision", "Recall", "F1"]
    a_vals = [p_a, r_a, f_a]
    b_vals = [p_b, r_b, f_b]

    x = np.arange(len(labels))
    width = 0.35

    plt.figure()
    plt.bar(x - width/2, a_vals, width, label=name_a)
    plt.bar(x + width/2, b_vals, width, label=name_b)
    plt.xticks(x, labels)
    plt.title(title)
    plt.ylabel("Score")
    plt.legend()
    plt.ylim(0, 1)
    out = os.path.join(FIG_DIR, "metric_bars.png")
    plt.savefig(out, bbox_inches="tight", dpi=150)
    plt.show()
    print("Saved:", out)

# ---------- ISOLATION FOREST ----------
iso = IsolationForest(
    contamination=CONTAMINATION,
    random_state=RANDOM_STATE,
    n_estimators=IF_N_ESTIMATORS,
)
iso.fit(X)

# Higher scores = more anomalous
iso_scores = -iso.decision_function(X)

# Choose an operating point: top q% as fraud (align with contamination as a starting point)
q = CONTAMINATION * 100.0
y_pred_iso = preds_from_top_q(iso_scores, q=q)

# Metrics
prec_i = precision_score(y, y_pred_iso, zero_division=0)
rec_i = recall_score(y, y_pred_iso, zero_division=0)
f1_i = f1_score(y, y_pred_iso, zero_division=0)
aucpr_i = average_precision_score(y, iso_scores)
auc_i = roc_auc_score(y, iso_scores)

print("Isolation Forest metrics (top {:.3f}%):".format(q))
print(f"Precision={prec_i:.4f}, Recall={rec_i:.4f}, F1={f1_i:.4f}, AP={aucpr_i:.4f}, ROC-AUC={auc_i:.4f}")

# Plots
plot_iso_score_histogram(iso_scores)
plot_pr_curve(y, iso_scores, title_prefix="Isolation Forest")
plot_roc(y, iso_scores, title_prefix="Isolation Forest")
plot_confusion(y, y_pred_iso, title="IF Confusion (top-q%)", fname="if_confusion.png")
precision_at_k_curve(y, iso_scores, ks=(50,100,250,500,1000,2000,5000))

# ---------- DBSCAN (with PCA for speed) ----------
pca = PCA(n_components=PCA_COMPONENTS, random_state=RANDOM_STATE)
X_pca = pca.fit_transform(X)

db = DBSCAN(eps=DBSCAN_EPS, min_samples=DBSCAN_MIN_SAMPLES, n_jobs=-1)
db.fit(X_pca)
y_pred_db = np.array([1 if lab == -1 else 0 for lab in db.labels_])

# For PR/ROC, build a score proxy: anomalies get 1, others 0 (coarse, but useful for curves)
db_scores = y_pred_db.astype(float)

prec_d = precision_score(y, y_pred_db, zero_division=0)
rec_d = recall_score(y, y_pred_db, zero_division=0)
f1_d = f1_score(y, y_pred_db, zero_division=0)
ap_d = average_precision_score(y, db_scores)
auc_d = roc_auc_score(y, db_scores)

print("DBSCAN metrics (label -1 as anomaly):")
print(f"Precision={prec_d:.4f}, Recall={rec_d:.4f}, F1={f1_d:.4f}, AP={ap_d:.4f}, ROC-AUC={auc_d:.4f}")

# Plots
plot_confusion(y, y_pred_db, title="DBSCAN Confusion", fname="dbscan_confusion.png")
plot_pr_curve(y, db_scores, title_prefix="DBSCAN (proxy)")
plot_roc(y, db_scores, title_prefix="DBSCAN (proxy)")

# ---------- COMPARISON FIGURES + TIDY TABLE ----------
metric_bars(y, y_pred_iso, y_pred_db, name_a="IsolationForest", name_b="DBSCAN")

results = pd.DataFrame({
    "Model": ["IsolationForest", "DBSCAN"],
    "Precision": [prec_i, prec_d],
    "Recall": [rec_i, rec_d],
    "F1": [f1_i, f1_d],
    "Average Precision (AP)": [aucpr_i, ap_d],
    "ROC-AUC": [auc_i, auc_d],
    "Alerts (positives)": [int(np.sum(y_pred_iso)), int(np.sum(y_pred_db))],
    "Total": [len(y), len(y)]
})
print(results)
results_path = os.path.join(FIG_DIR, "results_metrics.csv")
results.to_csv(results_path, index=False)
print("Saved metrics table to:", results_path)
